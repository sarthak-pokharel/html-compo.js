See It In Action

Create reusable components with just HTML - no JavaScript frameworks required!

<!-- Define once -->
<html-components>
  <user-card compo-attrs="name,role">
    <div class="card">
      <h3>@{name}</h3>
      <p>@{role}</p>
    </div>
  </user-card>
</html-components>

<!-- Use everywhere -->
<user-card name="Alice" role="Developer"></user-card>
<user-card name="Bob" role="Designer"></user-card>
